	function f_cal_age (p_date, p_item, p_item2)
	{	
		var today = new Date();
		dateStr = today.getDate();
		monthStr = today.getMonth();
		yearStr = today.getFullYear();
		
		obj = document.getElementById(p_date);
		year = (obj.value.substr(6, 4)-543) ;
		month =  (obj.value.substr(3, 2)-1) ;
		date =obj.value.substr(0, 2);

		theYear = yearStr - year;
		theMonth = monthStr - month;
		theDate = dateStr - date;

		var days = "";
		 if (monthStr == 0 || monthStr == 2 || monthStr == 4 || monthStr == 6 || monthStr == 7 || monthStr == 9 || monthStr == 11) days = 31;
		 if (monthStr == 3 || monthStr == 5 || monthStr == 8 || monthStr == 10) days = 30;
		 if (monthStr == 1) days = 28;
	
		 inYears = theYear;
	
		 if (month < monthStr && date > dateStr) { 
			  inYears = parseInt(inYears) + 1;
			  inMonths = theMonth - 1; 
		 };
		 
		 if (month < monthStr && date <= dateStr) {
			  inMonths = theMonth; 
		 } else if (month == monthStr && (date < dateStr || date == dateStr)) { 
			  inMonths = 0; 
		 } else if (month == monthStr && date > dateStr) { 
			  inMonths = 11; 
		 } else if (month > monthStr && date <= dateStr) { 
			  inYears = inYears - 1;
			  inMonths = ((12 - -(theMonth)) + 1); 
		 } else if (month > monthStr && date > dateStr) { 
			  inMonths = ((12 - -(theMonth))); 
		 };
		 
		 if (date < dateStr) { 
			  inDays = theDate; 
		 } else if (date == dateStr) { 
			  inDays = 0; 
		 } else { 
			  inYears = inYears - 1; 
			  inDays = days - (-(theDate)); 
		 };

		document.getElementById(p_item).value = inYears;
		document.getElementById(p_item2).value = inMonths;

		 if ((inYears < 18) || (inYears >= 31))
		{
				document.getElementById(p_date).value = "";
				document.getElementById(p_item).value =0;
				document.getElementById(p_item2).value =0;
				//document.getElementById(p_date).focus();
				msgbox('alert','อายุไม่ได้ตามเกณฑ์ที่กำหนด !!');
				return false;
		}
	}

	function checkfile(id, p_browse , p_text , p_type)
	{
		var jpg = new Array('.jpeg','.JPEG','.jpg','.JPG','.gif','.GIF','.png','.PNG');
		var error = 0;
		for(var j=0;j<jpg.length;j++)
			{
				if(document.getElementById(id).value.indexOf(jpg[j]) == -1)
					{ 
					error++;
					}
			}
				if(error == jpg.length)
					{
					alert("เลือกได้เฉพาะไฟล์นามสุล jpg , gif หรือ png เท่านั้น");
					document.getElementById(p_browse).value='';
					document.getElementById(p_text).value='';
					document.getElementById(p_type).value='';
					document.getElementById(id).select();
					return false;
					} 
		
		return true;
	}

	function f_chk_queue(p_item , p_prj_id)
	{ 
     
		url = "prc_yproject_project_member_chk_queue.php?queue_date=" + p_item + "&prj_id="+p_prj_id ;
		//alert (url);
		if (window.XMLHttpRequest) 
		{ // Non-IE browsers 
		  req = new XMLHttpRequest(); 
		  req.onreadystatechange = targetDiv; 
		  try
		   { 
				req.open("GET", url, true); 
		 	}
		 	catch (e)
		 	 { 
				alert(e); 
		 	 } 
		  req.send(null); 
		}
		 else if (window.ActiveXObject)
		  { // IE 
			  req = new ActiveXObject("Microsoft.XMLHTTP"); 
			  if (req)
			   { 
					req.onreadystatechange = targetDiv; 
					req.open("GET", url, true); 
					req.send(); 		
			  } 
		} 
}
